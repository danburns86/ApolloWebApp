@code {
    private string IssueType = "Bunch";
    private int SelectedLockId = 0;

    public class IssueForm
    {
        [Range(1, int.MaxValue)] public int SelectedId { get; set; }
        [Required] public string IssuedTo { get; set; } = string.Empty;
    }

    public class IssuedItemDto
    {
        public bool IsBunch { get; set; }
        public int Id { get; set; }
        public string ItemName { get; set; } = "";
        public string IssuedToName { get; set; } = "";
        public DateTime IssueDate { get; set; }
    }

    public class KeyVerifyItem
    {
        public int KeyId { get; set; }
        public string KeyCode { get; set; } = "";
        public string LockName { get; set; } = "";
        public string Status { get; set; } = "Returned";
    }

    private IssueForm IssueModel = new();
    private List<Bunch> AvailableBunches = new();
    private List<Bunch> AllActiveBunches = new();
    private List<KeyRecord> AvailableKeys = new();
    private List<KeyRecord> FilteredAvailableKeys = new();
    private List<HardwareLock> LocksWithAvailableKeys = new();
    private List<IssuedItemDto> IssuedItems = new();

    private int? TransferItemId;
    private bool? TransferIsBunch;
    private string TransferToName = "";
    
    private bool ShowBunchingModal = false;
    private KeyRecord? KeyToBunch;
    private int TargetBunchId = 0;

    private bool ShowReturnModal = false;
    private IssuedItemDto? ItemToReturn;
    private List<KeyVerifyItem> VerificationList = new();

    protected override async Task OnInitializedAsync() => await LoadData();

    private async Task LoadData()
    {
        AvailableBunches = await Db.Bunches.Include(b => b.Department).Where(b => b.IsActive && !b.IsLost && string.IsNullOrEmpty(b.IssuedToName)).OrderBy(b => b.Name).ToListAsync();
        AllActiveBunches = await Db.Bunches.Where(b => b.IsActive && !b.IsLost).OrderBy(b => b.Name).ToListAsync();
        
        var attachedKeyIds = await Db.KeyBunchAssignments.Where(ka => ka.RemovedDate == null).Select(ka => ka.KeyRecordId).ToListAsync();
        AvailableKeys = await Db.Keys.Include(k => k.Lock).Where(k => k.Status == "In Stock" && string.IsNullOrEmpty(k.IssuedToName) && !attachedKeyIds.Contains(k.Id)).ToListAsync();
        LocksWithAvailableKeys = AvailableKeys.Where(k => k.Lock != null).Select(k => k.Lock!).DistinctBy(l => l.Id).OrderBy(l => l.LockCode).ToList();

        IssuedItems.Clear();
        var outBunches = await Db.Bunches.Where(b => b.IsActive && !string.IsNullOrEmpty(b.IssuedToName)).ToListAsync();
        foreach (var b in outBunches) IssuedItems.Add(new IssuedItemDto { IsBunch = true, Id = b.Id, ItemName = b.Name, IssuedToName = b.IssuedToName ?? "Unknown", IssueDate = b.IssueDate ?? DateTime.Now });
        
        var outKeys = await Db.Keys.Include(k => k.Lock).Where(k => !string.IsNullOrEmpty(k.IssuedToName)).ToListAsync();
        foreach (var k in outKeys) IssuedItems.Add(new IssuedItemDto { IsBunch = false, Id = k.Id, ItemName = $"{(k.Lock?.LockCode ?? "??")} {k.SequenceNumber}", IssuedToName = k.IssuedToName ?? "Unknown", IssueDate = DateTime.Now });

        IssuedItems = IssuedItems.OrderByDescending(i => i.IssueDate).ToList();
        if (SelectedLockId != 0) FilterKeys();
    }

    private void OnLockSelected(ChangeEventArgs e) { if (int.TryParse(e.Value?.ToString(), out int id)) { SelectedLockId = id; FilterKeys(); } }
    private void FilterKeys() => FilteredAvailableKeys = AvailableKeys.Where(k => k.LockId == SelectedLockId).OrderBy(k => k.SequenceNumber).ToList();

    private async Task<string> GetStaffDisplayName()
    {
        var authState = await AuthStateProvider.GetAuthenticationStateAsync();
        var user = authState.User;
        if (user.Identity?.IsAuthenticated == true)
        {
            var appUser = await Db.Users.FirstOrDefaultAsync(u => u.UserName == user.Identity.Name);
            if (appUser != null && !string.IsNullOrWhiteSpace(appUser.FirstName)) return $"{appUser.FirstName} {appUser.LastName}".Trim();
        }
        return user.Identity?.Name ?? "System";
    }

    private async Task IssueItem()
    {
        string staffName = await GetStaffDisplayName();
        if (IssueType == "Bunch")
        {
            var bunch = await Db.Bunches.Include(b => b.KeyAssignments!).ThenInclude(ka => ka.KeyRecord).FirstOrDefaultAsync(b => b.Id == IssueModel.SelectedId);
            if (bunch != null)
            {
                bunch.IssuedToName = IssueModel.IssuedTo; bunch.IssueDate = DateTime.Now; bunch.ReturnDate = null;
                Db.CheckoutTransactions.Add(new CheckoutTransaction { BunchId = bunch.Id, IssuedToName = IssueModel.IssuedTo, IssueDate = DateTime.Now, AuthorizedByUserName = staffName });
                foreach (var ka in bunch.KeyAssignments!.Where(a => a.RemovedDate == null)) { if (ka.KeyRecord != null) ka.KeyRecord.Status = "Issued"; }
            }
        }
        else
        {
            var key = await Db.Keys.FirstOrDefaultAsync(k => k.Id == IssueModel.SelectedId);
            if (key != null)
            {
                key.IssuedToName = IssueModel.IssuedTo; key.Status = "Issued";
                Db.CheckoutTransactions.Add(new CheckoutTransaction { KeyRecordId = key.Id, IssuedToName = IssueModel.IssuedTo, IssueDate = DateTime.Now, AuthorizedByUserName = staffName });
            }
        }
        await Db.SaveChangesAsync();
        IssueModel = new(); SelectedLockId = 0; await LoadData();
    }

    private async Task StartReturnVerification(IssuedItemDto item)
    {
        ItemToReturn = item;
        VerificationList.Clear();
        if (item.IsBunch)
        {
            var bunch = await Db.Bunches.Include(b => b.KeyAssignments!).ThenInclude(ka => ka.KeyRecord).ThenInclude(k => k!.Lock).FirstOrDefaultAsync(b => b.Id == item.Id);
            if (bunch != null)
            {
                foreach (var ka in bunch.KeyAssignments!.Where(a => a.RemovedDate == null))
                {
                    VerificationList.Add(new KeyVerifyItem { KeyId = ka.KeyRecordId, KeyCode = $"{(ka.KeyRecord?.Lock?.LockCode ?? "")} {ka.KeyRecord?.SequenceNumber}", LockName = ka.KeyRecord?.Lock?.Name ?? "" });
                }
            }
        }
        else
        {
            VerificationList.Add(new KeyVerifyItem { KeyId = item.Id, KeyCode = item.ItemName, Status = "Returned" });
        }
        ShowReturnModal = true;
    }

    private void CloseReturnModal() { ShowReturnModal = false; ItemToReturn = null; }

    private async Task ConfirmReturn()
    {
        if (ItemToReturn == null) return;
        string staffName = await GetStaffDisplayName();
        if (ItemToReturn.IsBunch)
        {
            var b = await Db.Bunches.Include(x => x.KeyAssignments!).ThenInclude(ka => ka.KeyRecord).FirstOrDefaultAsync(x => x.Id == ItemToReturn.Id);
            if (b != null)
            {
                var trans = await Db.CheckoutTransactions.FirstOrDefaultAsync(t => t.BunchId == b.Id && t.ReturnDate == null);
                if (trans != null) trans.ReturnDate = DateTime.Now;
                foreach (var verify in VerificationList)
                {
                    var ka = b.KeyAssignments!.FirstOrDefault(a => a.KeyRecordId == verify.KeyId && a.RemovedDate == null);
                    var key = ka?.KeyRecord;
                    if (verify.Status == "Returned") { if (key != null) { key.Status = "In Stock"; key.IssuedToName = null; key.LastSeenDate = DateTime.Now; } }
                    else
                    {
                        if (ka != null) ka.RemovedDate = DateTime.Now;
                        if (key != null)
                        {
                            key.Status = (verify.Status == "Lost") ? "Lost" : "Issued";
                            key.IsLostLockout = (verify.Status == "Lost");
                            if (verify.Status == "Missing")
                            {
                                key.IssuedToName = ItemToReturn.IssuedToName;
                                Db.CheckoutTransactions.Add(new CheckoutTransaction { KeyRecordId = key.Id, IssuedToName = ItemToReturn.IssuedToName, IssueDate = ItemToReturn.IssueDate, AuthorizedByUserName = staffName });
                            }
                            else { key.IssuedToName = null; }
                        }
                    }
                }
                b.IssuedToName = null; b.ReturnDate = DateTime.Now;
            }
        }
        else
        {
            var k = await Db.Keys.FirstOrDefaultAsync(x => x.Id == ItemToReturn.Id);
            if (k != null)
            {
                var verify = VerificationList.First();
                k.Status = (verify.Status == "Returned") ? "In Stock" : (verify.Status == "Lost" ? "Lost" : "Issued");
                if (verify.Status == "Returned") { k.IssuedToName = null; k.LastSeenDate = DateTime.Now; }
                var trans = await Db.CheckoutTransactions.FirstOrDefaultAsync(t => t.KeyRecordId == k.Id && t.ReturnDate == null);
                if (trans != null) trans.ReturnDate = DateTime.Now;
            }
        }
        await Db.SaveChangesAsync();
        CloseReturnModal(); await LoadData();
    }

    private void StartTransfer(IssuedItemDto item) { TransferItemId = item.Id; TransferIsBunch = item.IsBunch; TransferToName = ""; }
    private void CancelTransfer() { TransferItemId = null; TransferIsBunch = null; }
    private async Task ConfirmTransfer(IssuedItemDto item)
    {
        if (string.IsNullOrWhiteSpace(TransferToName)) return;
        string staffName = await GetStaffDisplayName();
        if (item.IsBunch)
        {
            var b = await Db.Bunches.FirstOrDefaultAsync(x => x.Id == item.Id);
            if (b != null) { b.IssuedToName = TransferToName; b.IssueDate = DateTime.Now; }
        }
        else
        {
            var k = await Db.Keys.FirstOrDefaultAsync(x => x.Id == item.Id);
            if (k != null) k.IssuedToName = TransferToName;
        }
        var old = await Db.CheckoutTransactions.FirstOrDefaultAsync(t => (item.IsBunch ? t.BunchId == item.Id : t.KeyRecordId == item.Id) && t.ReturnDate == null);
        if (old != null) old.ReturnDate = DateTime.Now;
        Db.CheckoutTransactions.Add(new CheckoutTransaction { BunchId = item.IsBunch ? item.Id : null, KeyRecordId = item.IsBunch ? null : item.Id, IssuedToName = TransferToName, IssueDate = DateTime.Now, AuthorizedByUserName = staffName });
        await Db.SaveChangesAsync();
        CancelTransfer(); await LoadData();
    }

    private void OpenBunchingModal(KeyRecord k) { KeyToBunch = k; TargetBunchId = 0; ShowBunchingModal = true; }
    private void CloseBunchingModal() { ShowBunchingModal = false; KeyToBunch = null; }
    private async Task ConfirmAddToBunch()
    {
        if (KeyToBunch == null || TargetBunchId == 0) return;
        Db.KeyBunchAssignments.Add(new KeyBunchAssignment { BunchId = TargetBunchId, KeyRecordId = KeyToBunch.Id, AssignedDate = DateTime.Now });
        KeyToBunch.Status = "Issued"; await Db.SaveChangesAsync();
        CloseBunchingModal(); await LoadData();
    }
