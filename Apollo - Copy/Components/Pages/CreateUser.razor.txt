<PageTitle>Create System User</PageTitle>

<div class="d-flex justify-content-between align-items-center mb-3">
    <div>

    <a href="/admin/users" class="btn btn-outline-secondary shadow-sm">Back to Users</a>

<div class="row">
    <div class="col-md-8 col-lg-6">
        <div class="card shadow-sm border-primary">
            <div class="card-header bg-primary text-white fw-bold">
                Account Details
            </div>
            <div class="card-body">
                @if (!string.IsNullOrEmpty(ErrorMessage))
                {
                    <div class="alert alert-danger shadow-sm">
                        <strong><i class="bi bi-exclamation-triangle"></i> Error:</strong> @ErrorMessage
                    </div>
                }

                <EditForm Model="NewUserModel" OnValidSubmit="HandleCreateUser" FormName="AdminCreateUser">
                    <DataAnnotationsValidator />

@code {
    public class CreateUserForm
    {
        [Required(ErrorMessage = "Email is required.")]
        [EmailAddress(ErrorMessage = "Please enter a valid email address.")]
        public string Email { get; set; } = string.Empty;

        [Required(ErrorMessage = "Password is required.")]
        [MinLength(6, ErrorMessage = "Password must be at least 6 characters long.")]
        public string Password { get; set; } = string.Empty;

        public string InitialRole { get; set; } = string.Empty;
    }

    public CreateUserForm NewUserModel { get; set; } = new();
    private List<string> AvailableRoles = new();
    private string ErrorMessage = string.Empty;

    protected override async Task OnInitializedAsync()
    {
        AvailableRoles = await RoleManager.Roles.Select(r => r.Name!).ToListAsync();
    }

    private async Task HandleCreateUser()
    {
        ErrorMessage = string.Empty;

        // 1. Set up the new user
        var user = new ApplicationUser
        {
            UserName = NewUserModel.Email,
            Email = NewUserModel.Email,
            EmailConfirmed = true // Automatically confirm so they don't have to click an email link
        };

        // 2. Attempt to create them in the database
        var result = await UserManager.CreateAsync(user, NewUserModel.Password);

        if (result.Succeeded)
        {
            // 3. Assign role if selected
            if (!string.IsNullOrEmpty(NewUserModel.InitialRole))
            {
                await UserManager.AddToRoleAsync(user, NewUserModel.InitialRole);
            }

            // 4. Send the Admin back to the User Management dashboard
            Nav.NavigateTo("/admin/users");
        }
        else
        {
            // If the password was too weak or the email already exists, grab the errors
            ErrorMessage = string.Join(" ", result.Errors.Select(e => e.Description));
        }
    }
}
