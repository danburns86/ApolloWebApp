@code {
    private List<MaintenanceIssue> allIssues = new();
    private bool isLoading = true;

    protected override async Task OnInitializedAsync()
    {
        isLoading = true;
        allIssues = await MaintenanceService.GetAllIssuesAsync();
        isLoading = false;
    }

    private async Task PrintReport() => await JS.InvokeVoidAsync("window.print");
}
