<div class="container-fluid px-4 py-4 max-w-7xl">

            <h2 class="fw-bold mb-0">Maintenance Hub</h2>

    </div>

        <div class="text-center py-5"><div class="spinner-border text-danger"></div></div>
    }

@if (selectedIssue != null)

@code {
    private List<MaintenanceIssue> issues = new();
    private bool isLoading = true;

    private MaintenanceIssue? selectedIssue;
    private SnagStatus newStatus;
    private string adminNotes = "";

    protected override async Task OnInitializedAsync()
    {
        await LoadSnags();
    }

    private async Task LoadSnags()
    {
        isLoading = true;
        issues = await MaintenanceService.GetPendingSnags();
        isLoading = false;
    }

    private string? assignedTo = "";
    private string? contractorName = "";
    private string? quoteRef = "";

    private void OpenTriageModal(MaintenanceIssue issue) // Called OpenTriageModal in Dashboard.razor
    {
        selectedIssue = issue;
        newStatus = issue.Status;
        adminNotes = issue.AdminNotes ?? "";
        assignedTo = issue.AssignedTo;
        contractorName = issue.ContractorCompany;
        quoteRef = issue.QuoteReference;
    }

    private void CloseModal()
    {
        selectedIssue = null;
    }

    private async Task SaveTriage() // Called SaveTriage in Dashboard.razor
    {
        if (selectedIssue != null)
        {
            await MaintenanceService.UpdateSnagStatus(selectedIssue.Id, newStatus, adminNotes, assignedTo, contractorName, quoteRef); await LoadSnags();
            CloseModal();
        }
    }

    // --- VISUAL HELPERS ---
    private string GetSeverityBorderClass(SnagSeverity severity) => severity switch
    {
        SnagSeverity.Emergency => "border-danger",
        SnagSeverity.High => "border-warning",
        SnagSeverity.Medium => "border-primary",
        _ => "border-secondary"
    };

    private string GetSeverityBadgeClass(SnagSeverity severity) => severity switch
    {
        SnagSeverity.Emergency => "bg-danger",
        SnagSeverity.High => "bg-warning text-dark",
        SnagSeverity.Medium => "bg-primary",
        _ => "bg-secondary"
    };

    private string GetSeverityBgClass(SnagSeverity severity) => severity switch
    {
        SnagSeverity.Emergency => "bg-danger",
        SnagSeverity.High => "bg-warning text-dark",
        _ => "bg-dark"
    };
}
