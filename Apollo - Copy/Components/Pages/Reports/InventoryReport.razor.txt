@code {
    private string SearchTerm = "";
    private string SelectedHolder = "";
    private string StatusFilter = "";

    private List<string> AllHoldersList = new();
    private Dictionary<HardwareLock, List<KeyRecord>> LockGroups = new();
    private Dictionary<int, string> KeysInBunches = new();
    private Dictionary<int, KeyHolderInfo> ActiveKeyHolders = new();

    public class KeyHolderInfo { public string HolderName { get; set; } = ""; }

    protected override async Task OnInitializedAsync()
    {
        await LoadData();
    }

    private async Task LoadData()
    {
        using var context = await DbFactory.CreateDbContextAsync();

        AllHoldersList.Clear();
        ActiveKeyHolders.Clear();
        KeysInBunches.Clear();
        var holderNamesTracker = new HashSet<string>();

        // 1. Get all keys with related data
        var allKeys = await context.Keys
            .Include(k => k.Lock)
            .Include(k => k.StorageLocation)
            .ToListAsync();

        LockGroups = allKeys
            .Where(k => k.Lock != null)
            .GroupBy(k => k.Lock!)
            .ToDictionary(g => g.Key, g => g.ToList());

        // 2. Map Keys to Bunches
        var activeAssignments = await context.KeyBunchAssignments
            .Include(ka => ka.Bunch)
            .Where(ka => ka.RemovedDate == null)
            .ToListAsync();

        foreach (var ka in activeAssignments)
        {
            if (ka.KeyRecordId != 0 && ka.Bunch != null)
            {
                KeysInBunches[ka.KeyRecordId] = ka.Bunch.Name;
                // If a bunch is issued, the key is effectively issued to that person
                if (!string.IsNullOrEmpty(ka.Bunch.IssuedToName))
                {
                    ActiveKeyHolders[ka.KeyRecordId] = new KeyHolderInfo { HolderName = ka.Bunch.IssuedToName };
                    holderNamesTracker.Add(ka.Bunch.IssuedToName);
                }
            }
        }

        // 3. Track storage locations and individual holders
        foreach (var key in allKeys)
        {
            if (key.StorageLocation != null) holderNamesTracker.Add(key.StorageLocation.Name);
            if (!string.IsNullOrEmpty(key.IssuedToName))
            {
                holderNamesTracker.Add(key.IssuedToName);
                if (!ActiveKeyHolders.ContainsKey(key.Id))
                    ActiveKeyHolders[key.Id] = new KeyHolderInfo { HolderName = key.IssuedToName };
            }
        }

        AllHoldersList = holderNamesTracker.OrderBy(n => n).ToList();
    }

    private List<KeyRecord> GetFilteredKeys(List<KeyRecord> keys)
    {
        var query = keys.AsEnumerable();

        if (!string.IsNullOrWhiteSpace(SearchTerm))
            query = query.Where(k => (k.Lock?.LockCode ?? "").Contains(SearchTerm, StringComparison.OrdinalIgnoreCase) ||
                                     (k.Lock?.Name ?? "").Contains(SearchTerm, StringComparison.OrdinalIgnoreCase));

        if (!string.IsNullOrWhiteSpace(StatusFilter))
            query = query.Where(k => k.Status == StatusFilter);

        if (!string.IsNullOrWhiteSpace(SelectedHolder))
        {
            query = query.Where(k =>
                (k.StorageLocation?.Name == SelectedHolder) ||
                (k.IssuedToName == SelectedHolder) ||
                (ActiveKeyHolders.ContainsKey(k.Id) && ActiveKeyHolders[k.Id].HolderName == SelectedHolder)
            );
        }

        return query.ToList();
    }

    private async Task ExportToCsv()
    {
        var sb = new System.Text.StringBuilder();
        sb.AppendLine("Lock,Seq,Status,Holder/Location,Assignment,Last Seen");

        foreach (var group in LockGroups)
        {
            foreach (var key in group.Value)
            {
                var holder = !string.IsNullOrEmpty(key.IssuedToName) ? key.IssuedToName :
                            (ActiveKeyHolders.ContainsKey(key.Id) ? ActiveKeyHolders[key.Id].HolderName :
                            (key.StorageLocation?.Name ?? "Cabinet"));
                
                var assignment = KeysInBunches.ContainsKey(key.Id) ? KeysInBunches[key.Id] : "Loose";

                sb.AppendLine($"{group.Key.LockCode},{group.Key.LockCode} {key.SequenceNumber},\"{key.Status}\",\"{holder}\",\"{assignment}\",\"{key.LastSeenDate}\"");
            }
        }

        var fileBytes = System.Text.Encoding.UTF8.GetBytes(sb.ToString());
        using var streamRef = new DotNetStreamReference(new System.IO.MemoryStream(fileBytes));
        await JS.InvokeVoidAsync("downloadFileFromStream", $"Inventory_Report_{DateTime.Now:yyyyMMdd}.csv", streamRef);
    }
