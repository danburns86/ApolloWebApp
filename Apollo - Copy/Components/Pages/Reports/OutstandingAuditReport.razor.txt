@code {
    private string GroupBy = "Person"; // Default

    public class PrintItemDto
    {
        public string ItemName { get; set; } = "";
        public string HolderName { get; set; } = "";
        public string DepartmentName { get; set; } = "";
        public DateTime IssueDate { get; set; }
        public bool IsBunch { get; set; }
        public string LockName { get; set; } = "";
        public List<string> ContainedKeys { get; set; } = new();
    }

    private List<PrintItemDto> AllOutstandingItems = new();
    private Dictionary<string, List<PrintItemDto>> GroupedItems = new();

    protected override async Task OnInitializedAsync()
    {
        await LoadData();
    }

    private async Task LoadData()
    {
        using var context = await DbFactory.CreateDbContextAsync();
        
        // Grab all active checkouts to get exact issue dates and current holders
        var activeCheckouts = await context.CheckoutTransactions
            .Include(t => t.Bunch).ThenInclude(b => b!.Department)
            .Include(t => t.Bunch).ThenInclude(b => b!.KeyAssignments!).ThenInclude(ka => ka.KeyRecord).ThenInclude(k => k!.Lock)
            .Include(t => t.KeyRecord).ThenInclude(k => k!.Lock)
            .Where(t => t.ReturnDate == null)
            .ToListAsync();

        AllOutstandingItems.Clear();

        foreach (var tx in activeCheckouts)
        {
            if (tx.Bunch != null)
            {
                var keys = tx.Bunch.KeyAssignments?
                    .Where(ka => ka.RemovedDate == null && ka.KeyRecord != null)
                    .Select(ka => $"{ka.KeyRecord!.Lock?.LockCode} {ka.KeyRecord.SequenceNumber} - {ka.KeyRecord.Lock?.Name}")
                    .ToList() ?? new List<string>();

                AllOutstandingItems.Add(new PrintItemDto
                {
                    ItemName = tx.Bunch.Name,
                    HolderName = tx.IssuedToName,
                    DepartmentName = tx.Bunch.Department?.Name ?? "General / Unassigned",
                    IssueDate = tx.IssueDate,
                    IsBunch = true,
                    ContainedKeys = keys
                });
            }
            else if (tx.KeyRecord != null)
            {
                AllOutstandingItems.Add(new PrintItemDto
                {
                    ItemName = $"{tx.KeyRecord.Lock?.LockCode} {tx.KeyRecord.SequenceNumber}",
                    HolderName = tx.IssuedToName,
                    DepartmentName = "Individual Checkouts (Loose)", // Loose keys don't belong to a department
                    IssueDate = tx.IssueDate,
                    IsBunch = false,
                    LockName = tx.KeyRecord.Lock?.Name ?? ""
                });
            }
        }

        OrganizeData();
    }

    private void OrganizeData()
    {
        if (GroupBy == "Person")
        {
            GroupedItems = AllOutstandingItems
                .GroupBy(i => i.HolderName)
                .ToDictionary(g => g.Key, g => g.ToList());
        }
        else // By Department
        {
            GroupedItems = AllOutstandingItems
                .GroupBy(i => i.DepartmentName)
                .ToDictionary(g => g.Key, g => g.ToList());
        }
    }
}
