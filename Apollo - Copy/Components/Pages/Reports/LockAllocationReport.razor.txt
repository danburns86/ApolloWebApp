@code {
    private List<HardwareLock> Locks = new();

    protected override async Task OnInitializedAsync()
    {
        using var context = await DbFactory.CreateDbContextAsync();
        Locks = await context.Locks
            .Include(l => l.Keys)
            .Where(l => l.IsActive)
            .OrderBy(l => l.LockCode)
            .ToListAsync();
    }
}
