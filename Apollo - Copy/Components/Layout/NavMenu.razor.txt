@code {
    private bool KeysExpanded = true;
    private string UserDisplayName = "User";
    
    // Fixed typo from ConfiOneg back to Config
    private PortalConfiguration? Config;

    [CascadingParameter]
    private Task<AuthenticationState> AuthStateTask { get; set; } = default!;

    protected override async Task OnInitializedAsync()
    {
        using var db = await DbFactory.CreateDbContextAsync();
        
        Config = await db.PortalConfigurations.FirstOrDefaultAsync();

        var authState = await AuthStateTask;
        var user = authState.User;

        if (user?.Identity is { IsAuthenticated: true, Name: not null })
        {
            var appUser = await db.Users.FirstOrDefaultAsync(u => u.UserName == user.Identity.Name);
            if (appUser != null && !string.IsNullOrWhiteSpace(appUser.FirstName))
            {
                UserDisplayName = $"{appUser.FirstName} {appUser.LastName}".Trim();
            }
            else
            {
                UserDisplayName = user.Identity.Name;
            }
        }
    }

    private void ToggleKeys() => KeysExpanded = !KeysExpanded;
}
