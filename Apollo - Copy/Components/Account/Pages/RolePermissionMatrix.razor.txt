<div class="d-flex justify-content-between align-items-center mb-3">

    <div class="row mb-4">

                    <select class="form-select border-primary" @bind="SelectedRoleId" @bind:after="LoadRoleClaims">

            @if (!string.IsNullOrEmpty(SelectedRoleId))

                <button class="btn btn-success btn-lg shadow-sm" @onclick="SavePermissions">
                    <i class="bi bi-floppy"></i> Save Permissions for @SelectedRoleName

@code {
    private List<SystemPermission> AllPermissions = new();
    private List<IdentityRole> Roles = new();
    
    private string SelectedRoleId = string.Empty;
    private string SelectedRoleName = string.Empty;
    
    // This hashset holds the "ClaimValue" strings (e.g. "Keys.IssueReturn") that are currently ticked
    private HashSet<string> SelectedClaims = new();
    
    private bool ShowSuccessMessage = false;

    protected override async Task OnInitializedAsync()
    {
        Roles = await RoleManager.Roles.OrderBy(r => r.Name).ToListAsync();
        await LoadPermissions();
    }

    private async Task LoadPermissions()
    {
        AllPermissions = await Db.SystemPermissions.OrderBy(p => p.Module).ThenBy(p => p.Name).ToListAsync();
    }

    private async Task LoadRoleClaims()
    {
        ShowSuccessMessage = false;
        SelectedClaims.Clear();
        SelectedRoleName = string.Empty;

        if (string.IsNullOrEmpty(SelectedRoleId)) return;

        var role = await RoleManager.FindByIdAsync(SelectedRoleId);
        if (role != null)
        {
            SelectedRoleName = role.Name ?? "Unknown Role";
            
            // Get all claims for this role
            var claims = await RoleManager.GetClaimsAsync(role);
            
            // Add the ones that are "Permission" claims to our checked list
            foreach (var claim in claims.Where(c => c.Type == "Permission"))
            {
                SelectedClaims.Add(claim.Value);
            }
        }
    }

    private void TogglePermission(string claimValue, object? isChecked)
    {
        ShowSuccessMessage = false;
        if ((bool)(isChecked ?? false))
        {
            SelectedClaims.Add(claimValue);
        }
        else
        {
            SelectedClaims.Remove(claimValue);
        }
    }

    private async Task SavePermissions()
    {
        if (string.IsNullOrEmpty(SelectedRoleId)) return;

        var role = await RoleManager.FindByIdAsync(SelectedRoleId);
        if (role != null)
        {
            // 1. Get existing permission claims
            var existingClaims = await RoleManager.GetClaimsAsync(role);
            var existingPermissions = existingClaims.Where(c => c.Type == "Permission").ToList();

            // 2. Remove all of them to give us a clean slate
            foreach (var claim in existingPermissions)
            {
                await RoleManager.RemoveClaimAsync(role, claim);
            }

            // 3. Add the ones that are currently checked in the UI
            foreach (var claimValue in SelectedClaims)
            {
                await RoleManager.AddClaimAsync(role, new Claim("Permission", claimValue));
            }

            ShowSuccessMessage = true;
            _ = Task.Delay(3000).ContinueWith(_ => InvokeAsync(() => { ShowSuccessMessage = false; StateHasChanged(); }));
        }
    }

    // --- UI HELPER ---
    private string GetHeaderColor(string moduleName)
    {
        return moduleName switch
        {
            "Key Management" => "bg-primary text-white",
            "User Management" => "bg-success text-white",
            "Admin Functions" => "bg-danger text-white",
            "Facilities" => "bg-info text-dark", // New
            "Maintenance" => "bg-warning text-dark", // New
            _ => "bg-secondary text-white"
        };
    }

    // --- SEED DEFAULT DATA ---
    private async Task SeedDefaultPermissions()
    {
        var defaults = new List<SystemPermission>
        {
            // Existing
            new() { Module = "Key Management", Name = "View Key Inventory", ClaimValue = "Inventory.View" },
            new() { Module = "Key Management", Name = "Generate New Keys", ClaimValue = "Keys.Create" },
            new() { Module = "Key Management", Name = "Manage Bunches", ClaimValue = "Bunches.Manage" },
            new() { Module = "Key Management", Name = "Issue & Return Items", ClaimValue = "Transactions.IssueReturn" },
            new() { Module = "Key Management", Name = "Perform Formal Audits", ClaimValue = "Audit.Perform" },
            new() { Module = "User Management", Name = "View Staff Profiles", ClaimValue = "Users.View" },
            new() { Module = "User Management", Name = "Assign Roles", ClaimValue = "Users.ManageRoles" },
            new() { Module = "Admin Functions", Name = "Manage Core Settings", ClaimValue = "Admin.ManageSettings" },
            new() { Module = "Admin Functions", Name = "Full System Access", ClaimValue = "Admin.FullAccess" },

            // NEW: Facilities
            new() { Module = "Facilities", Name = "View Facilities", ClaimValue = "Facilities.View" },
            new() { Module = "Facilities", Name = "Manage Areas & Rooms", ClaimValue = "Facilities.Manage" },

            // NEW: Maintenance
            new() { Module = "Maintenance", Name = "View Maintenance Book", ClaimValue = "Maintenance.View" },
            new() { Module = "Maintenance", Name = "Manage/Approve Snags", ClaimValue = "Maintenance.Manage" },
            new() { Module = "Maintenance", Name = "View Contractor Reports", ClaimValue = "Maintenance.ContractorView" }
        };

        // Only add ones that don't exist yet
        foreach (var d in defaults)
        {
            if (!Db.SystemPermissions.Any(p => p.ClaimValue == d.ClaimValue))
            {
                Db.SystemPermissions.Add(d);
            }
        }

        await Db.SaveChangesAsync();
        await LoadPermissions();
    }
}
